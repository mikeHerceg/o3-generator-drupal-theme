import { Canvas, Story } from '@storybook/addon-docs';

# Select

<Canvas>
    <Story id="forms-select--default"/>
</Canvas>
<div class="documentation__container">
  <div class="documentation__col">
    <h3>Usage</h3>
    <p>
      The select input element allows users to enter data into applications and websites. All select input elements should include the following states:
      <ul>
      <li>Hover state</li>
      <li>Focus state</li>
      <li>Disabled state</li>
      <li>Error state</li>
      </ul>
    </p>
    <h3>Labelling Expectations</h3>
    <ul>
      <li>All select input elements should be associated with a <code>label</code>.</li>
    </ul>
    <h3>Focus Expectations</h3>
    <ul>
      <li>All select input elements should have a visible keyboard focus state.</li>
    </ul>
  </div>
  <div class="documentation__col">
    <h3>Keyboard Expectations</h3>
    <ul class="documentation__keys">
      <li><code>Tab</code> = Move to next focusable form element</li>
      <li><code>Enter</code> or <code>Space</code> = Open select dropdown</li>
      <li><code>&uparrow;</code> <code>&downarrow;</code> = Cycle through dropdown items</li>
      <li><code>Return</code> = Select dropdown item</li>
    </ul>
    <h3>Screen Reader Expectations</h3>
    <p>
      When the select input element is focused, screen readers should announce the following:
      <ul class="documentation__keys">
        <li>Select input label</li>
        <li>Select dropdown items</li>
      </ul>
    </p>
  </div>
</div>
